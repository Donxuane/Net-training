create table customers(
id int8 generated by default as identity primary key,
tablenumber int4,
food int4,
amount int4,
constraint customer_menu_fk_food_id foreign key (food) references menu(id)
);

insert into customers (tablenumber,food,amount)
values
(7,null,1);

insert into menu(foodname, price )
values
('kababi',6);

select 
c.tablenumber ,
m.foodname ,
c.amount ,
m.price ,
c.amount * m.price as totalprice
from customers c
left join menu m on c.food = m.id;

create view joinofcustomersmenu as
select 
c.tablenumber ,
m.foodname ,
c.amount ,
m.price ,
c.amount * m.price as totalprice
from customers c
left join menu m on c.food = m.id;


select 
*
from joinofcustomersmenu;
